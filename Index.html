Kode Google Apps Script (Copy ke script.google.com)

// === BUAT SPREADSHEET DENGAN 2 SHEET === // Sheet 1: "Master" — kolom: barcode | nama | kategori | satuan | shelf_life_days // Sheet 2: "Expired" — kolom: barcode | nama | kategori | jumlah | satuan | tgl_masuk | tgl_expired | lokasi | catatan | created_at function doGet(e) { var action = e.parameter.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); // === LOOKUP SATU BARCODE === if (action === 'lookup') { var barcode = e.parameter.barcode || ''; var master = ss.getSheetByName('Master'); var data = master.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === String(barcode).trim()) { return ContentService.createTextOutput(JSON.stringify({ found: true, barcode: data[i][0], namaProduk: data[i][1], kategori: data[i][2] || '', satuan: data[i][3] || 'pcs', shelfLife: parseInt(data[i][4]) || 0 })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput( JSON.stringify({found:false}) ).setMimeType(ContentService.MimeType.JSON); } // === SYNC SEMUA MASTER === if (action === 'sync') { var master = ss.getSheetByName('Master'); var data = master.getDataRange().getValues(); var items = []; for (var i = 1; i < data.length; i++) { if (data[i][0] && data[i][1]) { items.push({ barcode: String(data[i][0]).trim(), namaProduk: data[i][1], kategori: data[i][2] || '', satuan: data[i][3] || 'pcs', shelfLife: parseInt(data[i][4]) || 0 }); } } return ContentService.createTextOutput( JSON.stringify({items: items, total: items.length}) ).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput('OK'); } // === TERIMA DATA PRODUK EXPIRED === function doPost(e) { var sheet = ss.getSheetByName('Expired'); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.barcode, data.namaProduk, data.kategori, data.jumlah, data.satuan, data.tanggalMasuk, data.tanggalExpired, data.lokasi, data.catatan, new Date().toLocaleString('id-ID') ]); return ContentService.createTextOutput( JSON.stringify({status:'ok'}) ).setMimeType(ContentService.MimeType.JSON); } // Deploy: New deployment > Web app // Execute as: Me | Who has access: Anyone
